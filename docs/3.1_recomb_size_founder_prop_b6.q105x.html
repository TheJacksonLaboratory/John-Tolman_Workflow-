  mat = matrix(0, nrow = nrow(probs) * dim(probs)[3], ncol = 8,
               dimnames = list(rep(rownames(probs), dim(probs)[3]),
                               names(CCcolors)))
  for(i in 1:dim(probs)[3]) {
    st = (i - 1) * nrow(probs) + 1
    en = i * nrow(probs)
    mat[st:en,] = probs[,,i]
  } # for(i)
  
  return(data.frame(chr = rep(markers$chr, each = nrow(probs)), 
                    pos = rep(markers$pos, each = nrow(probs)), mat))
  
} # reformat_probs()

{
  chr <- map_tab[, chr_col]
  if (is.null(chr_names))
    chr_names <- unique(chr)
  chr <- factor(chr, levels = chr_names)

  map_tab[order(chr, pos, seq_len(nrow(map_tab))), , drop = FALSE]
}

split_map <- function (map, chr_names = NULL)
{
  map <- reorder_map_table(map, chr_names = chr_names)
  pos <- as.numeric(map[, 2])
  chr <- map[, 1]
  uchr <- unique(chr)
  names(pos) <- rownames(map)
  lapply(split(pos, factor(chr, uchr)), sort)
}
